use std::time::Instant;
use ark_serialize::CanonicalSerialize;
fn run_hyrax_once(n: usize) -> (u128, u128, u128, usize) {

    let chacha = &mut ChaCha20Rng::from_rng(test_rng()).unwrap();

    let pp = Hyrax381::setup(1, Some(n), chacha).unwrap();

    let (ck, vk) = Hyrax381::trim(&pp, 1, 1, None).unwrap();

    let l_poly = LabeledPolynomial::new(
        "test_poly".to_string(),
        rand_poly::<Fr>(0, Some(n), chacha),
        None,
        None,
    );

    let start_commit = Instant::now();
    let (c, rands) = Hyrax381::commit(&ck, &[l_poly.clone()], Some(chacha)).unwrap();
    let commit_ms = start_commit.elapsed().as_millis();

    let point: Vec<Fr> = rand_point(Some(n), chacha);
    let value = l_poly.evaluate(&point);

    // Dummy argument
    let mut test_sponge = test_sponge::<Fr>();

    let start_prover = Instant::now();
    let proof = Hyrax381::open(
        &ck,
        &[l_poly],
        &c,
        &point,
        &mut (test_sponge.clone()),
        &rands,
        Some(chacha),
    )
    .unwrap();
    let prover_ms = start_prover.elapsed().as_millis();

    //verify
    let start_verify = Instant::now();
    assert!(Hyrax381::check(
        &vk,
        &c,
        &point,
        [value],
        &proof,
        &mut test_sponge,
        Some(chacha),
    )
    .unwrap());
    let verifier_ms = start_verify.elapsed().as_millis();

    // proof size
    let mut proof_bytes = Vec::new();
    proof.serialize_compressed(&mut proof_bytes).unwrap();
    let proof_size = proof_bytes.len();

    // let g1_elems = n;
    // // BLS12-381 compressed G1 size
    // let g1_bytes = 48;
    // let proof_size = g1_elems * g1_bytes;




    (commit_ms, prover_ms, verifier_ms, proof_size)
}

type LeafH = LeafIdentityHasher;
    type CompressH = Sha256;
    type ColHasher<F, D> = FieldToBytesColHasher<F, D>;

    struct MerkleTreeParams;

    impl Config for MerkleTreeParams {
        type Leaf = Vec<u8>;

        type LeafDigest = <LeafH as CRHScheme>::Output;
        type LeafInnerDigestConverter = ByteDigestConverter<Self::LeafDigest>;
        type InnerDigest = <CompressH as TwoToOneCRHScheme>::Output;

        type LeafHash = LeafH;
        type TwoToOneHash = CompressH;
    }

    type MTConfig = MerkleTreeParams;

    type BrakedownPCS<F> = LinearCodePCS<
        MultilinearBrakedown<F, MTConfig, SparseMultilinearExtension<F>, ColHasher<F, Blake2s256>>,
        F,
        SparseMultilinearExtension<F>,
        MTConfig,
        ColHasher<F, Blake2s256>,
    >;

fn rand_poly<Fr: PrimeField>(
        _: usize,
        num_vars: Option<usize>,
        rng: &mut ChaCha20Rng,
    ) -> SparseMultilinearExtension<Fr> {
        match num_vars {
            Some(n) => SparseMultilinearExtension::rand(n, rng),
            None => unimplemented!(), // should not happen in ML case!
        }
    }

fn rand_point<F: Field>(num_vars: Option<usize>, rng: &mut ChaCha20Rng) -> Vec<F> {
        match num_vars {
            Some(n) => (0..n).map(|_| F::rand(rng)).collect(),
            None => unimplemented!(), // should not happen!
        }
    }

fn run_brakedown_once(log_n: usize) -> (u128, u128, u128, usize) {
    use ark_crypto_primitives::crh::{CRHScheme, TwoToOneCRHScheme};

    let n = 1 << log_n;
    let mut rng = test_rng();

    // === setup parameters ===
    let leaf_hash_param = <LeafH as CRHScheme>::setup(&mut rng).unwrap();
    let two_to_one_hash_param =
        <CompressH as TwoToOneCRHScheme>::setup(&mut rng).unwrap();
    let col_hash_params =
        <ColHasher<Fr, Blake2s256> as CRHScheme>::setup(&mut rng).unwrap();

    let pp: BrakedownPCParams<Fr, MTConfig, ColHasher<Fr, Blake2s256>> =
        BrakedownPCParams::default(
            &mut rng,
            n,
            true,
            leaf_hash_param,
            two_to_one_hash_param,
            col_hash_params,
        );

    let (ck, vk) = BrakedownPCS::<Fr>::trim(&pp, 0, 0, None).unwrap();

    // === random multilinear polynomial ===
    let mut chacha = ChaCha20Rng::from_rng(test_rng()).unwrap();
    let poly = rand_poly::<Fr>(0, Some(log_n), &mut chacha);

    let labeled_poly = LabeledPolynomial::new(
        "poly".to_string(),
        poly.clone(),
        Some(log_n),
        Some(log_n),
    );

    //commit
    let start_commit = Instant::now();

    let (commitment, states) =
        BrakedownPCS::<Fr>::commit(&ck, &[labeled_poly.clone()], None).unwrap();
    
    let commit_ms = start_commit.elapsed().as_millis();

    let point = rand_point::<Fr>(Some(log_n), &mut chacha);
    let value = poly.evaluate(&point);
    let mut sponge = test_sponge::<Fr>();

    let start_prove = Instant::now();
    let proof = BrakedownPCS::<Fr>::open(
        &ck,
        &[labeled_poly],
        &commitment,
        &point,
        &mut (sponge.clone()),
        &states,
        None,
    )
    .unwrap();

    let prove_ms = start_prove.elapsed().as_millis();

    // === verify ===
    let start_verify = Instant::now();
    BrakedownPCS::<Fr>::check(
        &vk,
        &commitment,
        &point,
        [value],
        &proof,
        &mut sponge,
        None,
    )
    .unwrap();
    let verifier_ms = start_verify.elapsed().as_millis();

    //proof size

    let mut proof_bytes = Vec::new();
    proof.serialize_compressed(&mut proof_bytes).unwrap();
    let proof_size = proof_bytes.len();
    // let proof_elements = 1 << (log_n / 2);
    // let proof_size = proof_elements * 1000;


    (commit_ms, prove_ms, verifier_ms, proof_size)
}
